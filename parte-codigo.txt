
    function calcula(num) {
            if (typeof gdisplay == 'undefined') {
                document.calcForm.display.value = '' ;
            }
        document.calcForm.display.value += num;
        gdisplay = 1;
    }

    function limpa() {
        document.calcForm.display.value = '';
        delete gvalor, gop, gdisplay;
    }

    function operacao(op, valor1, valor2) {
        if (op == 'soma') {
            var valor = parseFloat(valor1) + parseFloat(valor2) ;
        } else 
        if (op == 'subtrai') {
            var valor = parseFloat(valor1) - parseFloat(valor2) ;
        } else
        if (op == 'multi') {
            var valor = parseFloat(valor1) * parseFloat(valor2) ;
        } else 
        if (op == 'divide') {
            var valor = parseFloat(valor1) / parseFloat(valor2) ;
        }
        
        return (valor);
    }

    function calcParse(op) {
        var valor = document.calcForm.display.value;
        delete gdisplay;

            if (typeof gop != 'undefined' && op == 'calcula') {
                gvalor = operacao(gop, gvalor, valor);
                document.calcForm.display.value = gvalor;
                
                delete op;
                delete gvalor;
                return (0);
            }
        
            if (typeof gvalor != 'undefined') {
                gvalor = operacao(gop, gvalor, valor);
                gop = op;
                document.calcForm.display.value = gvalor;
            } else {
                gvalor = valor;
                gop = op;
            }
    }
